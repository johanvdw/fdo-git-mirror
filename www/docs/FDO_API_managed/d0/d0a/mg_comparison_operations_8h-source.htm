00036 /// set ([^abcdef]).
00037 ///

00040 {
00041     /// Test to see if the left and right expressions are equal.
00042     ComparisonOperations_EqualTo = FdoComparisonOperations_EqualTo,

00044     /// Test to see if the left and right expressions are not equal.
00045     ComparisonOperations_NotEqualTo = FdoComparisonOperations_NotEqualTo,
00046 
00047     /// Test to see if the left expression is greater than the right expression.
00048     ComparisonOperations_GreaterThan = FdoComparisonOperations_GreaterThan,
00049

00051     ComparisonOperations_GreaterThanOrEqualTo = FdoComparisonOperations_GreaterThanOrEqualTo,

00054     ComparisonOperations_LessThan = FdoComparisonOperations_LessThan,
00055

00057     ComparisonOperations_LessThanOrEqualTo = FdoComparisonOperations_LessThanOrEqualTo,
